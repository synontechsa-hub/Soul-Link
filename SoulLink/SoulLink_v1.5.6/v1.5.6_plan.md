**Why first:** Every stage after this reads system data from the DB. This table must exist before any seeder runs.

### 1.1 — Create `SystemConfig` model

**File:** `backend/app/models/system_config.py` — **NEW FILE**

```python
# /backend/app/models/system_config.py
from sqlmodel import SQLModel, Field
from sqlalchemy import JSON
from datetime import datetime, timezone
from typing import Dict, Any, Optional

class SystemConfig(SQLModel, table=True):
    """
    Key-value store for all system JSON configurations.
    Replaces all _dev/data/system/*.json runtime reads.
    One row per config file.
    """
    __tablename__ = "system_config"

    key: str = Field(primary_key=True, max_length=100)
    data: Dict[str, Any] = Field(default_factory=dict, sa_type=JSON)
    version: str = Field(default="1.5.6", max_length=20)
    updated_at: datetime = Field(
        default_factory=lambda: datetime.now(timezone.utc))
```

### 1.3 — Run table migration

## Stage 3 — Build Seeder Scripts

### 3.1 — Create `seed_system_v156.py`

**File:** `_dev/scripts/seed_system_v156.py` — **NEW FILE**

Reads all 10 system JSONs → upserts into `system_config` table.

```python
"""
seed_system_v156.py
Seeds all system JSON configs into the system_config DB table.
Usage:
    python seed_system_v156.py           # Live run
    python seed_system_v156.py --dry-run # Preview only
"""

import sys
import json
from pathlib import Path

SCRIPT_DIR = Path(__file__).resolve().parent
PROJECT_ROOT = SCRIPT_DIR.parent.parent
sys.path.insert(0, str(PROJECT_ROOT))

from backend.app.database.session import engine
from backend.app.models.system_config import SystemConfig
from sqlmodel import Session

SYSTEM_DIR = PROJECT_ROOT / "_dev" / "data" / "system"
DRY_RUN = "--dry-run" in sys.argv

# Map: DB key → filename
SYSTEM_FILES = {
    "weather":        "weather.json",
    "time_and_day":   "time_and_day.json",
    "calendar":       "calendar.json",
    "archetypes":     "archetypes.json",
    "factions":       "factions.json",
    "routines":       "routines.json",
    "intimacy":       "intimacy.json",
    "relationships":  "relationships.json",
    "monetization":   "monetization.json",
    "global_config":  "global_config.json",
}

def seed():
    print(f"\n{'[DRY RUN] ' if DRY_RUN else ''}System Config Seeder v1.5.6")
    success, failed = 0, 0

    with Session(engine) as session:
        for key, filename in SYSTEM_FILES.items():
            fpath = SYSTEM_DIR / filename
            if not fpath.exists():
                print(f"  SKIP {key} — {filename} not found")
                continue
            try:
                with open(fpath, "r", encoding="utf-8") as f:
                    data = json.load(f)

                if DRY_RUN:
                    print(f"  [PREVIEW] {key} ← {filename}")
                    success += 1
                    continue

                existing = session.get(SystemConfig, key)
                if existing:
                    existing.data = data
                    existing.version = data.get("version", "1.5.6")
                    session.add(existing)
                else:
                    session.add(SystemConfig(
                        key=key,
                        data=data,
                        version=data.get("version", "1.5.6")
                    ))

                session.commit()
                print(f"  OK {key}")
                success += 1
            except Exception as e:
                session.rollback()
                print(f"  FAIL {key} → {e}")
                failed += 1

    print(f"\nDone → {success} seeded, {failed} failed")

if __name__ == "__main__":
    seed()
```

### 3.2 — Create `seed_lore_v156.py`

**File:** `_dev/scripts/seed_lore_v156.py` — **NEW FILE**

Reads `system/lore.json` → upserts into `lore_items` table.

```python
"""
seed_lore_v156.py
Seeds lore knowledge items into the lore_items DB table.
Usage:
    python seed_lore_v156.py           # Live run
    python seed_lore_v156.py --dry-run # Preview only
"""

import sys
import json
from pathlib import Path
from datetime import datetime, timezone

SCRIPT_DIR = Path(__file__).resolve().parent
PROJECT_ROOT = SCRIPT_DIR.parent.parent
sys.path.insert(0, str(PROJECT_ROOT))

from backend.app.database.session import engine
from backend.app.models.lore_item import LoreItem
from sqlmodel import Session

LORE_FILE = PROJECT_ROOT / "_dev" / "data" / "system" / "lore.json"
DRY_RUN = "--dry-run" in sys.argv

def seed():
    print(f"\n{'[DRY RUN] ' if DRY_RUN else ''}Lore Seeder v1.5.6")

    with open(LORE_FILE, "r", encoding="utf-8") as f:
        data = json.load(f)

    items = data.get("knowledge_items", [])
    print(f"Found {len(items)} lore items\n")
    success, failed = 0, 0

    with Session(engine) as session:
        for item in items:
            item_id = item.get("id")
            if not item_id:
                continue
            try:
                if DRY_RUN:
                    print(f"  [PREVIEW] {item_id} [{item.get('category')}]")
                    success += 1
                    continue

                existing = session.get(LoreItem, item_id)
                lore_data = dict(
                    id=item_id,
                    category=item.get("category", "general"),
                    topics=item.get("topics", []),
                    content=item.get("content", {}),
                    associations=item.get("associations", {}),
                    source_metadata={"source": "lore.json", "version": "1.5.6"},
                    version="1.5.6",
                    updated_at=datetime.now(timezone.utc)
                )

                if existing:
                    for k, v in lore_data.items():
                        setattr(existing, k, v)
                    session.add(existing)
                else:
                    session.add(LoreItem(**lore_data))

                session.commit()
                print(f"  OK {item_id}")
                success += 1
            except Exception as e:
                session.rollback()
                print(f"  FAIL {item_id} → {e}")
                failed += 1

    print(f"\nDone → {success} seeded, {failed} failed")

if __name__ == "__main__":
    seed()
```

### 3.3 — Create master seeder `seed_all_v156.py`

**File:** `_dev/scripts/seed_all_v156.py` — **NEW FILE**

One command to seed everything in dependency order.

```python
"""
seed_all_v156.py
Master seeder — runs all seeders in correct dependency order.
Usage:
    python seed_all_v156.py           # Live run
    python seed_all_v156.py --dry-run # Preview all, no DB writes
"""

import sys
import subprocess
from pathlib import Path

SCRIPT_DIR = Path(__file__).resolve().parent
DRY_RUN = "--dry-run" in sys.argv
ARGS = ["--dry-run"] if DRY_RUN else []

SEEDERS = [
    ("System Config",  SCRIPT_DIR / "seed_system_v156.py"),
    ("Locations",      SCRIPT_DIR / "seed_locations_v156.py"),
    ("Souls",          SCRIPT_DIR / "seed_souls_v156.py"),
    ("Lore Items",     SCRIPT_DIR / "seed_lore_v156.py"),
]

print(f"\n{'=' * 50}")
print(f"  SoulLink v1.5.6 — Master Seeder")
print(f"  Mode: {'DRY RUN' if DRY_RUN else 'LIVE'}")
print(f"{'=' * 50}\n")

for name, script in SEEDERS:
    print(f"\n▶ Running: {name}")
    print("-" * 40)
    result = subprocess.run(
        [sys.executable, str(script)] + ARGS,
        capture_output=False
    )
    if result.returncode != 0:
        print(f"\n❌ {name} failed with exit code {result.returncode}")
        print("Stopping seed run. Fix errors and re-run.")
        sys.exit(result.returncode)

print(f"\n{'=' * 50}")
print(f"  ✅ All seeders complete")
print(f"{'=' * 50}\n")
```
